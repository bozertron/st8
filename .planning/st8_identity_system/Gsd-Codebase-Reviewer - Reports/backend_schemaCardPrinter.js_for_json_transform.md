        console.error('[st8:printer] Schema cards directory not found:', schemaCardsDir);

    }

    const files = fs.readdirSync(schemaCardsDir).filter(f => f.endsWith('.json'));
    let printed = 0;
    let errors = 0;

    for (const file of files) {
        try {
            const card = JSON.parse(fs.readFileSync(path.join(schemaCardsDir, file), 'utf-8'));
            const result = this.printCard(card);
            if (result !== null) {
                printed++;
            }

            console.error(`[st8:printer] Error printing ${file}:`, err.message);
            errors++;
        }
    }

    console.log(`[st8:printer] Printed ${printed} cards, ${errors} errors`);

    const pruned = this.pruneOldCards();

## SUMMARY OF FINDINGS

### BLOCKER Issues

**B-01: notificationBus → printer path is dead code (notificationBus.js:62-67)**
The `notificationBus` was designed to call `printer.printCard(event.schemaCard)` for live .txt updates. However, **no caller in the entire codebase passes `schemaCard` in the publish event**. Every `notificationBus.publish()` call sends events with only `{fingerprint, filepath, mutationType, actor, sha256Hash}` — never a `schemaCard` object. The condition `if (this.printer && event.schemaCard)` at notificationBus.js:62 is always false. This means:
- The "human-readable fallback" described in the docblock is non-functional for live updates
- .txt cards are ONLY generated during the initial startup run
- All incremental file changes (CREATE, EDIT, DELETE, LOCK, CONCEPT, PRODUCTION) are invisible in .txt output

### WARNING Issues

**W-01: .txt cards never updated during watch mode (index.js:193-401)**
The watcher callback handles file mutations and emits JSON schema cards via `emitter.emitCard()`, but never calls `printer.printCard()` or `printer.printAllFromCards()`. Combined with B-01, `.txt` files become stale immediately after startup. The `LATEST_` files (42 on disk) never reflect changes after initial indexing.

**W-02: 413 stale .archive .txt files never cleaned up**
Files from `.archive/` paths were printed before the guard (lines 58-63) was added. `pruneOldCards()` only limits count per base name (max 10), so files for unique `.archive` base names persist indefinitely. There is no `cleanupIgnoredFiles()` method to retroactively remove them.

**W-03: LATEST_ files for deleted source files are orphaned forever**
When a source file is deleted, its JSON schema card is removed from `.st8/schema-cards/`, but the `LATEST_{name}.txt` file is never deleted. `pruneOldCards()` explicitly skips `LATEST_` files (line 206). Orphaned `LATEST_` files will accumulate indefinitely.

**W-04: Return value type mismatch in `printAllFromCards()` (line 288-290)**
`this.pruneOldCards()` returns `{ pruned: number, kept: number }`, but `printAllFromCards()` returns it as the `pruned` field, making the return `{ printed, errors, pruned: { pruned, kept } }`. A caller expecting `result.pruned` to be a number would get an object. Currently no caller uses the return value, but this is a latent bug.
